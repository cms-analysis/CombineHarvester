81  boost::replace_all(subbed_name, "$BIN", proc.bin());

83  boost::replace_all(subbed_name, "$MASS", proc.mass());
84  boost::replace_all(subbed_name, "$ERA", proc.era());
85  boost::replace_all(subbed_name, "$CHANNEL", proc.channel());
86  boost::replace_all(subbed_name, "$ANALYSIS", proc.analysis());
87  auto sys = std::make_shared<Systematic>();
88  ch::SetProperties(sys.get(), &proc);
89  sys->set_name(subbed_name);
90  sys->set_type(type);
91  if (type == "lnN" || type == "lnU") {
92  sys->set_asymm(asymm);
93  sys->set_value_u(val_u);
94  sys->set_value_d(val_d);
95  } else if (type == "shape" || type == "shapeN2") {
96  sys->set_asymm(true);
97  sys->set_value_u(1.0);
98  sys->set_value_d(1.0);
99  sys->set_scale(val_u);
100  }
101  CreateParameterIfEmpty(sys->name());
102  if (sys->type() == "lnU") {
103  params_.at(sys->name())->set_err_d(0.);
104  params_.at(sys->name())->set_err_u(0.);
105  }
106  systs_.push_back(sys);
