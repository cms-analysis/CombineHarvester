188 double CombineHarvester::GetRateInternal(ProcSystMap const& lookup,
189  std::string const& single_sys) {
190  double rate = 0.0;
191  // TH1F tot_shape
192  for (unsigned i = 0; i < procs_.size(); ++i) {
193  double p_rate = procs_[i]->rate();
194  // If we are evaluating the effect of a single parameter
195  // check the list of associated nuisances and skip if
196  // this "single_sys" is not in the list
197  // However - we can't skip if the process has a pdf, as
198  // we haven't checked what the parameters are
199  if (single_sys != "" && !procs_[i]->pdf()) {
200  if (!ch::any_of(lookup[i], [&](Systematic const* sys) {
201  return sys->name() == single_sys;
202  })) continue;
203  }
204  for (auto sys_it : lookup[i]) {
205  double x = params_[sys_it->name()]->val();
206  if (sys_it->asymm()) {
207  p_rate *= logKappaForX(x * sys_it->scale(), sys_it->value_d(),
208  sys_it->value_u());
209  } else {
210  p_rate *= std::pow(sys_it->value_u(), x * sys_it->scale());
211  }
212  }
213  rate += p_rate;
214  }

344 void CombineHarvester::ShapeDiff(double x,
345  TH1F * target,
346  TH1 const* nom,
347  TH1 const* low,
348  TH1 const* high,
349  bool linear) {
350  double fx = smoothStepFunc(x);
351  for (int i = 1; i <= target->GetNbinsX(); ++i) {
352  float h = high->GetBinContent(i);
353  float l = low->GetBinContent(i);
354  float n = nom->GetBinContent(i);
355  if (!linear) {
356  float t = target->GetBinContent(i);
357  target->SetBinContent(i, t > 0. ? std::log(t) : -999.);
358  h = (h > 0. && n > 0.) ? std::log(h/n) : 0.;
359  l = (l > 0. && n > 0.) ? std::log(l/n) : 0.;
360  target->SetBinContent(i, target->GetBinContent(i) +
361  0.5 * x * ((h - l) + (h + l) * fx));
362  target->SetBinContent(i, std::exp(target->GetBinContent(i)));
363  } else {
364  target->SetBinContent(i, target->GetBinContent(i) +
365  0.5 * x * ((h - l) + (h + l - 2. * n) * fx));
366  }
367  }
368 }

370 void CombineHarvester::ShapeDiff(double x,
371  TH1F * target,
372  RooDataHist const* nom,
373  RooDataHist const* low,
374  RooDataHist const* high) {
375  double fx = smoothStepFunc(x);
376  for (int i = 1; i <= target->GetNbinsX(); ++i) {
377  high->get(i-1);
378  low->get(i-1);
379  nom->get(i-1);
380  // The RooDataHists are not scaled to unity (unlike in the TH1 case above)
381  // so we have to normalise the bin contents on the fly
382  float h = high->weight() / high->sumEntries();
383  float l = low->weight() / low->sumEntries();
384  float n = nom->weight() / nom->sumEntries();
385  target->SetBinContent(i, target->GetBinContent(i) +
386  0.5 * x * ((h - l) + (h + l - 2. * n) * fx));
387  }

473 void CombineHarvester::VariableRebin(std::vector<double> bins) {
474  // We need to keep a record of the Process rates before we rebin. The
475  // reasoning comes from the following scenario: the user might choose a new
476  // binning which excludes some of the existing bins - thus changing the
477  // process normalisation. This is fine, but we also need to adjust the shape
478  // Systematic entries - both the rebinning and the adjustment of the value_u
479  // and value_d shifts.
480  std::vector<double> prev_proc_rates(procs_.size());
481 
482  // Also hold on the scaled Process hists *after* the rebinning is done - these
483  // are needed to update the associated Systematic entries
484  std::vector<std::unique_ptr<TH1>> scaled_procs(procs_.size());
485 
486  for (unsigned i = 0; i < procs_.size(); ++i) {
487  if (procs_[i]->shape()) {
488  // Get the scaled shape here
489  std::unique_ptr<TH1> copy(procs_[i]->ClonedScaledShape());
490  // shape norm should only be "no_norm_rate"
491  prev_proc_rates[i] = procs_[i]->no_norm_rate();
492  std::unique_ptr<TH1> copy2(copy->Rebin(bins.size()-1, "", &(bins[0])));
493  // The process shape & rate will be reset here
494  procs_[i]->set_shape(std::move(copy2), true);
495  scaled_procs[i] = procs_[i]->ClonedScaledShape();
496  }
497  }
498  for (unsigned i = 0; i < obs_.size(); ++i) {
499  if (obs_[i]->shape()) {
500  std::unique_ptr<TH1> copy(obs_[i]->ClonedScaledShape());
501  std::unique_ptr<TH1> copy2(copy->Rebin(bins.size()-1, "", &(bins[0])));
502  obs_[i]->set_shape(std::move(copy2), true);
503  }
504  }
505  for (unsigned i = 0; i < systs_.size(); ++i) {
506  TH1 const* proc_hist = nullptr;
507  double prev_rate = 0.;
508  for (unsigned j = 0; j < procs_.size(); ++j) {
509  if (MatchingProcess(*(procs_[j]), *(systs_[i].get()))) {
510  proc_hist = scaled_procs[j].get();
511  prev_rate = prev_proc_rates[j];
512  }
513  }
514  if (systs_[i]->shape_u() && systs_[i]->shape_d()) {
515  // These hists will be normalised to unity
516  std::unique_ptr<TH1> copy_u(systs_[i]->ClonedShapeU());
517  std::unique_ptr<TH1> copy_d(systs_[i]->ClonedShapeD());
518 
519  // If we found a matching Process we will scale this back up to their
520  // initial rates
521  if (proc_hist) {
522  copy_u->Scale(systs_[i]->value_u() * prev_rate);
523  copy_d->Scale(systs_[i]->value_d() * prev_rate);
524  }
525  std::unique_ptr<TH1> copy2_u(
526  copy_u->Rebin(bins.size() - 1, "", &(bins[0])));
527  std::unique_ptr<TH1> copy2_d(
528  copy_d->Rebin(bins.size() - 1, "", &(bins[0])));
529  // If we have proc_hist != nullptr, set_shapes will re-calculate value_u
530  // and value_d for us, before scaling the new hists back to unity
531  systs_[i]->set_shapes(std::move(copy2_u), std::move(copy2_d), proc_hist);
532  }
533  }
534 }

555 // This implementation "borowed" from
556 // HiggsAnalysis/CombinedLimit/src/ProcessNormalization.cc
557 double CombineHarvester::logKappaForX(double x, double k_low,
558  double k_high) const {
559  if (k_high == 0. || k_low == 0.) {
560  if (verbosity_ >= 1) {
561  LOGLINE(log(), "Have kappa=0.0 (scaling ill-defined), returning 1.0");
562  }
563  return 1.;
564  }
565  if (fabs(x) >= 0.5)
566  return (x >= 0 ? std::pow(k_high, x) : std::pow(k_low, -1.0 * x));
567  // interpolate between log(kappaHigh) and -log(kappaLow)
568  // logKappa(x) = avg + halfdiff * h(2x)
569  // where h(x) is the 3th order polynomial
570  // h(x) = (3 x^5 - 10 x^3 + 15 x)/8;
571  // chosen so that h(x) satisfies the following:
572  // h (+/-1) = +/-1
