669 template <class Map>
670 void CombineHarvester::AddSyst(CombineHarvester& target,
671  std::string const& name, std::string const& type,
672  Map const& valmap) {
673  // Keep track of which Process entries get a Systematic assigned and which
674  // don't. If verbosity is on we'll print lists of these processes at the end.
675  std::vector<ch::Process *> not_added_procs;
676  std::vector<ch::Process *> added_procs;
677  // Also track which tuples in the map did not get used. Do this by getting the
678  // full map here and then removing elements as they are used to create a
679  // Systematic.
680  auto tuples = valmap.GetTupleSet();
681  if (verbosity_ >= 1) {
682  log() << (name + ":" + type) << "\n";
683  }
684  for (unsigned i = 0; i < procs_.size(); ++i) {
685  if (!valmap.Contains(procs_[i].get())) {
686  not_added_procs.push_back(procs_[i].get());
687  continue;
688  }
689  tuples.erase(valmap.GetTuple(procs_[i].get()));
690  added_procs.push_back(procs_[i].get());
691  double val_u = valmap.ValU(procs_[i].get());
692  double val_d = valmap.ValD(procs_[i].get());
693  std::string formula = valmap.Formula(procs_[i].get());
694  std::string args = valmap.Args(procs_[i].get());
695  target.AddSystFromProc(*(procs_[i]), name, type, valmap.IsAsymm(),
696  val_u, val_d, formula, args);
697  }
698  if (tuples.size() && verbosity_ >= 1) {
699  log() << ">> Map keys that were not used to create a Systematic:\n";
700  for (auto s : tuples) {
701  log() << ch::Tuple2String(s) << "\n";
702  }
703  }
704  if (verbosity_ >= 2) {
705  Process::PrintHeader(log());
706  log() << ">> Process entries that did not get a Systematic:\n";
707  for (auto p : not_added_procs) {
708  log() << *p << "\n";
709  }
710  log() << ">> Process entries that did get a Systematic:\n";
711  for (auto p : added_procs) {
712  log() << *p << "\n";
713  }
714  }
715 }
