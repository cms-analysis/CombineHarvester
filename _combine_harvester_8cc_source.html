651 std::shared_ptr<RooWorkspace> CombineHarvester::SetupWorkspace(
652  RooWorkspace const& ws, bool can_rename) {
653  bool name_in_use = false;
654 
655  // 1) Does the UUID of this ws match any other ws in the map?
656  for (auto const& it : wspaces_) {
657  // - Yes: just return a ptr to the matched ws
658  if (it.second->uuid() == ws.uuid() && flags_["workspace-uuid-recycle"]) {
659  FNLOGC(log(), verbosity_ >= 1)
660  << "Workspace with name " << it.second->GetName()
661  << " has the same UUID, will use this one\n";
662  return it.second;
663  }
664  if (!name_in_use && strcmp(it.second->GetName(), ws.GetName()) == 0) {
665  name_in_use = true;
666  }
667  }
668 
669  // 2) No: Ok will clone it in. Is the ws name already in use?
670  if (!name_in_use) {
671  // - No: clone with same name and return
672  // IMPORTANT: Don't used RooWorkspace::Clone(), it seems to introduce
673  // bugs
674  if (GetFlag("workspaces-use-clone")) {
675  wspaces_[std::string(ws.GetName())] = std::shared_ptr<RooWorkspace>(
676  reinterpret_cast<RooWorkspace*>(ws.Clone()));
677  } else {
678  wspaces_[std::string(ws.GetName())] =
679  std::make_shared<RooWorkspace>(RooWorkspace(ws));
680  }
681  return wspaces_.at(ws.GetName());
682  }
683 
684  // 3) Am I allowed to rename (default no)?
685  if (!can_rename) {
686  // - No: throw runtime error
687  throw std::runtime_error(FNERROR("A different workspace with name " +
688  std::string(ws.GetName()) +
689  " already exists"));
690  }
691 
692  // 4) Yes: determine first available nameX, clone, return
693  std::set<int> used_ints = {0};
694  std::string src_name(ws.GetName());
695  for (auto const& it : wspaces_) {
696  std::string test_name(it.second->GetName());
697  if (test_name.find(src_name) == 0) {
698  std::string postfix = test_name.substr(src_name.size());
699  try {
700  int number = boost::lexical_cast<int>(postfix);
701  used_ints.insert(number);
702  } catch (boost::bad_lexical_cast & e) {
703  }
704  }
705  }
706  std::string new_name =
707  src_name + boost::lexical_cast<std::string>(*(used_ints.rbegin()) + 1);
708  FNLOGC(log(), verbosity_ >= 1) << "Workspace with name " << src_name
709  << " already defined, renaming to " << new_name
710  << "\n";
711 
712  std::shared_ptr<RooWorkspace> new_wsp;
713  if (GetFlag("workspaces-use-clone")) {
714  new_wsp = std::shared_ptr<RooWorkspace>(
715  reinterpret_cast<RooWorkspace*>(ws.Clone(new_name.c_str())));
716  } else {
717  new_wsp = std::make_shared<RooWorkspace>(RooWorkspace(ws));
718  }
719  new_wsp->SetName(new_name.c_str());
720  wspaces_[new_name] = new_wsp;
721  return wspaces_.at(new_name);
722 }

724 void CombineHarvester::ImportParameters(RooArgSet *vars) {
725  auto x = vars->createIterator();
726  do {
727  RooRealVar *y = dynamic_cast<RooRealVar*>(**x);
728  if (y) {
729  if (!params_.count(y->GetName())) {
730  if (verbosity_ >= 1) {
731  log() << "[ImportParameters] Creating parameter from RooRealVar:\n";
732  y->printStream(log(), y->defaultPrintContents(0),
733  y->defaultPrintStyle(0), "[ImportParameters] ");
734  }
735  Parameter par;
736  par.set_name(y->GetName());
737  par.set_val(y->getVal());
738  if ((y->hasError() || y->hasAsymError()) &&
739  flags_["import-parameter-err"]) {
740  par.set_err_d(y->getErrorLo());
741  par.set_err_u(y->getErrorHi());
742  } else {
743  par.set_err_d(0.);
744  par.set_err_u(0.);
745  }
746  params_[y->GetName()] = std::make_shared<Parameter>(par);
747  } else {
748  if (verbosity_ >= 1)
749  log() << "[ImportParameters] Parameter \"" << y->GetName()
750  << "\" already exists\n";
751  }
752  Parameter *param = params_[y->GetName()].get();
753  std::vector<RooRealVar *> & p_vars = param->vars();
754  if (std::find(p_vars.begin(), p_vars.end(), y) == p_vars.end()) {
755  p_vars.push_back(y);
756  if (verbosity_ >= 1)
757  log() << "[ImportParameters] Parameter now stores " << p_vars.size()
758  << " link(s) to RooRealVar objects\n";
759  } else {
760  if (verbosity_ >= 1)
761  log() << "[ImportParameters] Parameter already stores link to "
762  "RooRealVar object\n";
763  }
764  }
765  } while (x->Next());
766 }

779 ch::Parameter* CombineHarvester::SetupRateParamVar(std::string const& name,
780  double val, bool is_ext_arg) {
781  RooWorkspace *ws = nullptr;
782  if (!wspaces_.count("_rateParams")) {
783  ws = this->SetupWorkspace(RooWorkspace("_rateParams","_rateParams")).get();
784  } else {
785  ws = wspaces_.at("_rateParams").get();
786  }
787  // Parameter doesn't exist in the workspace - let's create it
788  RooRealVar *var = ws->var(name.c_str());
789  if (!var) {
790  // The value doesn't matter, our Parameter object will set it later
791  RooRealVar tmp_var(name.c_str(), name.c_str(), 0);
792  ws->import(tmp_var);
793  var = ws->var(name.c_str());
794  FNLOGC(log(), verbosity_ > 1)
795  << "Created new RooRealVar for rateParam: " << name << "\n";
796  if (verbosity_ > 1) var->Print();
797  } else {
798  FNLOGC(log(), verbosity_ > 1)
799  << "Reusing existing RooRealVar for rateParam: " << name << "\n";
800  }
801  if (is_ext_arg) var->setAttribute("extArg");
802  Parameter * param = nullptr;
803  if (!params_.count(name)) {
804  params_[name] = std::make_shared<Parameter>(Parameter());
805  param = params_.at(name).get();
806  param->set_name(name);
807  param->set_err_u(0.);
808  param->set_err_d(0.);
809  } else {
810  param = params_.at(name).get();
811  }
812  // If the RooRealVar in the workpsace isn't in the list, add it
813  bool var_in_par = false;
814  for (auto const& ptr : param->vars()) {
815  if (ptr == var) {
816  var_in_par = true;
817  break;
818  }
819  }
820  if (!var_in_par) {
821  param->vars().push_back(var);
822  }
823  // Then this propagates the value to the RooRealVar
824  FNLOGC(log(), verbosity_ > 1) << "Updating parameter value from "
825  << param->val();
826  param->set_val(val);
827  if (verbosity_ > 1) log() << " to " << param->val() << "\n";
828  return param;
829 }
